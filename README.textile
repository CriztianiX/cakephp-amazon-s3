cakephp-amazon-s3
=======

Purpose
-------

Provides GET, PUT and DELETE functionality for CakePHP 1.3.x

Examples
--------

@<?= $this->Form->create('Pet', array('type' => 'file'));@
@<?= $this->Form->input('pet' , array('type' => 'file')); ?>@
@<?= $this->Form->end('Save my pooch'); ?>@
