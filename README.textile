h2. cakephp-amazon-s3

CakePHP 1.3.x component which provides RESTful GET, PUT and DELETE functionality for Amazon S3

h3. Examples

<pre>
// GET example
$this->AmazonS3->remote_object = 'WurcJ.jpg'; // Remote file to GET
$this->AmazonS3->local_dir = WWW_ROOT.'files/amazons3/test'; // Local dir where you want to save 
if($this->AmazonS3->get()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>

<pre>
// PUT example
$this->AmazonS3->local_object = 'subfolder/plain.txt'; // Local file to PUT. Prepend with any relevant dir's
$this->AmazonS3->local_dir = WWW_ROOT.'files/amazons3'; // Path to local dir which file resides
if($this->AmazonS3->put()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>


<pre>
// DELETE example
$this->AmazonS3->remote_object = 'folder_within_bucket/unwanted.txt'; // File on Amazon to DELETE
if($this->AmazonS3->delete()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>
