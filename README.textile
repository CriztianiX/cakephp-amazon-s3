h2. cakephp-amazon-s3

Provides GET, PUT and DELETE functionality for CakePHP 1.3.x

h3. Examples

<pre>
// GET example
$this->AmazonS3->remote_object = 'WurcJ.jpg'; // Thing to get
$this->AmazonS3->local_dir = WWW_ROOT.'files/amazons3/test'; //  /Users/rob/Projects/amz/app/webroot/files/
if($this->AmazonS3->get()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>

<pre>
// PUT example
$this->AmazonS3->local_object = 'subfolder/plain.txt';
$this->AmazonS3->local_dir = WWW_ROOT.'files/amazons3'; //  /Users/rob/Projects/amz/app/webroot/files/
if($this->AmazonS3->put()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>


<pre>
// DELETE example
$this->AmazonS3->remote_object = 'folder_within_bucket/unwanted.txt';
if($this->AmazonS3->delete()) {
	debug($this->AmazonS3->results);
} else {
	debug($this->AmazonS3->errors);
}
</pre>
